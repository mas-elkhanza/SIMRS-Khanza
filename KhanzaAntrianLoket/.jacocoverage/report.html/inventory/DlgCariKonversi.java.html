    private void isHitung(){
        if((!n1.getText().equals(""))&&(!n2.getText().equals(""))){
            if(!nmsat1.getText().equals(nmsat2.getText())){
                try {
                    ResultSet rs=koneksi.createStatement().executeQuery(
                            "select nilai,nilai_konversi from konver_sat where kode_sat='"+kdsat1.getSelectedItem()+"' and sat_konversi='"+kdsat2.getSelectedItem()+"'");
                    if(rs.next()){
                        n2.setText(Double.toString(Double.parseDouble(n1.getText())*(rs.getDouble(2)/rs.getDouble(1))));
                    }else if(!rs.next()){
                        int reply = JOptionPane.showConfirmDialog(rootPane,"Maaf, konversi satuan tidak ditemukan. Apa anda ingin menambahkan..??","Konfirmasi",JOptionPane.YES_NO_OPTION);
                        if (reply == JOptionPane.YES_OPTION) {
                            this.setCursor(Cursor.getPredefinedCursor(Cursor.WAIT_CURSOR));
                            DlgKonversi form=new DlgKonversi(null,false);
                            form.emptTeks();
                            form.tampil(" order by konver_sat.kode_sat");
                            form.setSize(internalFrame1.getWidth()-40,internalFrame1.getHeight()-40);
                            form.setLocationRelativeTo(internalFrame1);
                            form.setAlwaysOnTop(false);
                            form.setVisible(true);
                            this.setCursor(Cursor.getDefaultCursor());
                            dispose();
                        }                            
                    }                    
                } catch (SQLException ex) {
                    System.out.println(ex);
                }
            }
        }        
    }
