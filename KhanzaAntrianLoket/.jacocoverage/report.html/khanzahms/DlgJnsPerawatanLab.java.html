    private void BtnPrintActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_BtnPrintActionPerformed
        this.setCursor(Cursor.getPredefinedCursor(Cursor.WAIT_CURSOR));
        if(! TCari.getText().trim().equals("")){
            BtnCariActionPerformed(evt);
        }
        if(tabMode.getRowCount()==0){
            JOptionPane.showMessageDialog(null,"Maaf, data sudah habis. Tidak ada data yang bisa anda print...!!!!");
            BtnBatal.requestFocus();
        }else if(tabMode.getRowCount()!=0){
                Valid.MyReport("rptTarifLab.jrxml","report","::[ Data Tarif Laboratorium ]::","select jns_perawatan_lab.kd_jenis_prw,jns_perawatan_lab.nm_perawatan,jns_perawatan_lab.bagian_rs,"+
                    "jns_perawatan_lab.tarif_tindakan_petugas,jns_perawatan_lab.total_byr,penjab.png_jawab "+
                    "from jns_perawatan_lab inner join penjab on penjab.kd_pj=jns_perawatan_lab.kd_pj where "+
                    " jns_perawatan_lab.kd_jenis_prw like '%"+TCari.getText().trim()+"%' or  jns_perawatan_lab.nm_perawatan like '%"+TCari.getText().trim()+"%' or "+
                    " jns_perawatan_lab.bagian_rs like '%"+TCari.getText().trim()+"%' or jns_perawatan_lab.tarif_tindakan_petugas like '%"+TCari.getText().trim()+"%' or "+
                    " penjab.png_jawab like '%"+TCari.getText().trim()+"%' or jns_perawatan_lab.total_byr like '%"+TCari.getText().trim()+"%'  "+
                    "order by jns_perawatan_lab.kd_jenis_prw");   
        }
        this.setCursor(Cursor.getDefaultCursor());
}//GEN-LAST:event_BtnPrintActionPerformed

    public void tampil() {
        Valid.tabelKosong(tabMode);
        try{
            ps.setString(1,"%"+TCari.getText().trim()+"%");
            ps.setString(2,"%"+TCari.getText().trim()+"%");
            ps.setString(3,"%"+TCari.getText().trim()+"%");
            ps.setString(4,"%"+TCari.getText().trim()+"%");            
            ps.setString(5,"%"+TCari.getText().trim()+"%");
            ps.setString(6,"%"+TCari.getText().trim()+"%");

            rs=ps.executeQuery();
            while(rs.next()){
                String[] data={rs.getString(1),
                               rs.getString(2),
                               Valid.SetAngka(rs.getDouble(3)),
                               Valid.SetAngka(rs.getDouble(4)),
                               Valid.SetAngka(rs.getDouble(5)),
                               rs.getString(6)};
                tabMode.addRow(data);
            }
        }catch(SQLException e){
            System.out.println("Error : "+e);
        }
        LCount.setText(""+tabMode.getRowCount());
    }

    public void emptTeks() {
        TKd.setText("");
        TNm.setText("");
        kdpnj.setText("");
        nmpnj.setText("");
        BagianRs.setText("0");
        TTndPr.setText("0");
        TJmlPr.setText("0");
        //Valid.autoNomer(" jns_perawatan_lab ","JP",6,TKd);
        Valid.autoNomer3("select ifnull(MAX(CONVERT(RIGHT(kd_jenis_prw,3),signed)),0) from jns_perawatan_lab","J",6,TKd);
        TKd.requestFocus();
    }

    private void getData() {
        if(tbJnsPerawatan.getSelectedRow()!= -1){
            TKd.setText(tbJnsPerawatan.getValueAt(tbJnsPerawatan.getSelectedRow(),0).toString());
            TNm.setText(tbJnsPerawatan.getValueAt(tbJnsPerawatan.getSelectedRow(),1).toString());
            BagianRs.setText(tbJnsPerawatan.getValueAt(tbJnsPerawatan.getSelectedRow(),2).toString());
            TTndPr.setText(tbJnsPerawatan.getValueAt(tbJnsPerawatan.getSelectedRow(),3).toString());
            TJmlPr.setText(tbJnsPerawatan.getValueAt(tbJnsPerawatan.getSelectedRow(),4).toString());
            Sequel.cariIsi("select kd_pj from jns_perawatan_lab where kd_jenis_prw=?", kdpnj,tbJnsPerawatan.getValueAt(tbJnsPerawatan.getSelectedRow(),0).toString());
            nmpnj.setText(tbJnsPerawatan.getValueAt(tbJnsPerawatan.getSelectedRow(),5).toString());
        }
    }

    private void isjml(){
        DecimalFormat df2 = new DecimalFormat("####");
        if((! BagianRs.getText().equals(""))&&(! TTndPr.getText().equals(""))){
            double x=Double.parseDouble(BagianRs.getText().trim());
            double y=Double.parseDouble(TTndPr.getText().trim());
            TJmlPr.setText(df2.format(x+y));
        }
    }
