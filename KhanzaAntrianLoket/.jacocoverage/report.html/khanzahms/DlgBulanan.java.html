    public void tampil(String key) {
        String sql="select  pegawai.id, "+
                "pegawai.nik, "+
                "pegawai.nama, "+
                "rekap_presensi.shift, "+
                "rekap_presensi.jam_datang, "+
                "rekap_presensi.jam_pulang, "+
                "rekap_presensi.status, "+
                "rekap_presensi.keterlambatan, "+
                "rekap_presensi.durasi, "+
                "rekap_presensi.keterangan "+
                " from pegawai inner join rekap_presensi on pegawai.id=rekap_presensi.id where rekap_presensi.jam_datang like '%"+ThnCari.getSelectedItem()+"-"+BlnCari.getSelectedItem()+"%' order by pegawai.nama,rekap_presensi.jam_datang";
        prosesCari(sql);
    }

    private void prosesCari(String sql) {
        Valid.tabelKosong(tabMode);
        try{
            java.sql.Statement stat=connect.createStatement();
            ResultSet rs=stat.executeQuery(sql);
            rs.last();
            
            id=new String[rs.getRow()];
            tgl=new String[rs.getRow()];
            
            rs.beforeFirst();
            
            int i=0;
            
            while(rs.next()){
                String[] data={rs.getString(2),
                               rs.getString(3),
                               rs.getString(4),
                               rs.getString(5),
                               rs.getString(6),
                               rs.getString(7),
                               rs.getString(8),
                               rs.getString(9),
                               rs.getString(10)};
                tabMode.addRow(data);
                id[i]=rs.getString(1);
                tgl[i]=rs.getString(5);
                i++;
             }
            rs.close();
            stat.close();
        }catch(SQLException e){
            System.out.println("Error : "+e);
        }
        LCount.setText(""+tabMode.getRowCount());

    }

    private void getData() {
        int row=tbPegawai.getSelectedRow();
        if(row!= -1){
            no=row;
            Idpresensi.setText(id[no]);            
            Shift.setSelectedItem(tabMode.getValueAt(row,2).toString());            
            if(!tabMode.getValueAt(row,3).toString().equals("")){
                Jam.setSelectedItem(tabMode.getValueAt(row,3).toString().substring(11,13));
                Menit.setSelectedItem(tabMode.getValueAt(row,3).toString().substring(14,16));
                Detik.setSelectedItem(tabMode.getValueAt(row,3).toString().substring(17,19));
            }else if(tabMode.getValueAt(row,3).toString().equals("")){
                Jam.setSelectedItem("00");
                Menit.setSelectedItem("00");
                Detik.setSelectedItem("00");
            }            
                        
            if(!tabMode.getValueAt(row,4).toString().equals("")){
                Jam2.setSelectedItem(tabMode.getValueAt(row,4).toString().substring(11,13));
                Menit2.setSelectedItem(tabMode.getValueAt(row,4).toString().substring(14,16));
                Detik2.setSelectedItem(tabMode.getValueAt(row,4).toString().substring(17,19));
            }else if(tabMode.getValueAt(row,4).toString().equals("")){
                Jam2.setSelectedItem("00");
                Menit2.setSelectedItem("00");
                Detik2.setSelectedItem("00");
            }
            catatan.setText(tabMode.getValueAt(row,8).toString());
            
            try {
                tglMasuk.setDate(new SimpleDateFormat("yyyy-MM-dd").parse(
                                tabMode.getValueAt(row,3).toString().substring(0,4)+"-"+
                                tabMode.getValueAt(row,3).toString().substring(5,7)+"-"+
                                tabMode.getValueAt(row,3).toString().substring(8,10)));
            } catch (ParseException ex) {
                Logger.getLogger(DlgHarian.class.getName()).log(Level.SEVERE, null, ex);
            }
            
            try {
                tglPulang.setDate(new SimpleDateFormat("yyyy-MM-dd").parse(
                                tabMode.getValueAt(row,4).toString().substring(0,4)+"-"+
                                tabMode.getValueAt(row,4).toString().substring(5,7)+"-"+
                                tabMode.getValueAt(row,4).toString().substring(8,10)));
            } catch (ParseException ex) {
                Logger.getLogger(DlgHarian.class.getName()).log(Level.SEVERE, null, ex);
            }
        }        
    }
