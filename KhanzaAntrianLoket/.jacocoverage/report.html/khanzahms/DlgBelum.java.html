    public void tampil(String key) {
        String sql="select pegawai.nik,"+
                                "pegawai.nama,"+
                                "pegawai.jk,"+
                                "pegawai.jbtn,"+
                                "pegawai.jnj_jabatan,"+
                                "pegawai.departemen,"+
                                "pegawai.bidang "+
                                "from pegawai where pegawai.id not in (select temporary_presensi.id from temporary_presensi)";
        prosesCari(sql);
    }

    private void prosesCari(String sql) {
        Valid.tabelKosong(tabMode);
        try{
            java.sql.Statement stat=connect.createStatement();
            ResultSet rs=stat.executeQuery(sql);
            while(rs.next()){
                String[] data={rs.getString(1),
                               rs.getString(2),
                               rs.getString(3),
                               rs.getString(4),
                               rs.getString(5),
                               rs.getString(6),
                               rs.getString(7)};
                tabMode.addRow(data);
             }
            rs.close();
            stat.close();
        }catch(SQLException e){
            System.out.println("Error : "+e);
        }
        LCount.setText(""+tabMode.getRowCount());
    }
