    public void hapus(String pesan,String nm_table){
        int jawab=JOptionPane.showConfirmDialog(null, "Yakin anda mau menghapuas data "+pesan+" ????","Konfirmasi",JOptionPane.YES_NO_OPTION);
            if(jawab==JOptionPane.YES_OPTION){
                try{
                    java.sql.Statement statement=koneksi.createStatement();
                    statement.executeUpdate("delete from "+nm_table);
                }catch(SQLException e){
                    System.out.println("Pesan Error : "+e);
                    JOptionPane.showMessageDialog(null,"Maaf, Gagal dihapus. Ada sebagian data "+pesan+" yang masih dipakai di table lain...!!!!");
                }
            }
    }
