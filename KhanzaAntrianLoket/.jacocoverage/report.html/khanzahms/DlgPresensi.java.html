    private void BtnCariActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_BtnCariActionPerformed
        prosesCari("select  pegawai.id, "+
                "pegawai.nik, "+
                "pegawai.nama, "+
                "temporary_presensi.shift,"+
                "temporary_presensi.jam_datang, "+
                "temporary_presensi.jam_pulang, "+
                "temporary_presensi.status, "+
                "temporary_presensi.keterlambatan, "+
                "temporary_presensi.durasi "+
                " from pegawai inner join temporary_presensi on pegawai.id=temporary_presensi.id where "+
                   " pegawai.nik like '%"+TCari.getText().trim()+"%' "+
                   "or pegawai.nama like '%"+TCari.getText().trim()+"%' "+
                   "or temporary_presensi.shift like '%"+TCari.getText().trim()+"%' "+
                   "or temporary_presensi.status like '%"+TCari.getText().trim()+"%' "+
                   "or temporary_presensi.keterlambatan like '%"+TCari.getText().trim()+"%' "+
                   "or temporary_presensi.jam_datang like '%"+TCari.getText().trim()+"%' "+
                   "or temporary_presensi.jam_pulang like '%"+TCari.getText().trim()+"%' order by temporary_presensi.jam_datang ");
}//GEN-LAST:event_BtnCariActionPerformed

    private void BtnSimpanActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_BtnSimpanActionPerformed
        if(Bcd.getText().trim().equals("")){
            Valid.textKosong(Bcd,"ID");
        }else{
            try {
                Statement stat=connect.createStatement();
                ResultSet rs=stat.executeQuery("select * from barcode where barcode='"+Bcd.getText()+"'");
                rs.last();
                if(rs.getRow()>=1){
                        //----------------cari id Barcode-------------------------------------------
                        Sequel.cariIsi("select id from barcode where barcode='"+Bcd.getText()+"'", Idpresensi);
                        Sequel.cariIsi("select jam_jaga.jam_masuk from jam_jaga inner join pegawai on pegawai.departemen=jam_jaga.dep_id "+
                                " where jam_jaga.shift='"+Shift.getSelectedItem().toString()+"' and pegawai.id='"+Idpresensi.getText()+"'", jamdatang);
                        try {
                            //simpan ke temporary presensi
                            String jam="now()";
                            if(!jamdatang.getText().equals("")){
                                jam="CONCAT(CURRENT_DATE(),' "+jamdatang.getText()+"')";
                            }
                            stat.execute("insert into temporary_presensi values "+
                               "('"+Idpresensi.getText()+"','"+Shift.getSelectedItem()+"',NOW(),NULL,"+
                               "if(TIME_TO_SEC(now())-TIME_TO_SEC("+jam+")>600,'Terlambat','Tepat Waktu'),"+
                               "if(TIME_TO_SEC(now())-TIME_TO_SEC("+jam+")>600,SEC_TO_TIME(TIME_TO_SEC(now())-TIME_TO_SEC("+jam+")),''),'','')");
                        }catch (Exception ex) {
                            //System.out.println(ex);                            
                            try{
                                //update jam pulang dan durasi sebelum disimpan ke table rekap presensi
                                stat.execute("update temporary_presensi set jam_pulang=NOW(),"+
                                    "durasi=(SEC_TO_TIME(unix_timestamp(now()) - unix_timestamp(jam_datang)))"+
                                    " where id='"+Idpresensi.getText()+"'");
                                //cari dulu data di temporary yang mau disimpan ke rekap presensi
                                ResultSet rs2=stat.executeQuery("select * from temporary_presensi where id='"+Idpresensi.getText()+"'");
                                if(rs2.next()){
                                    try{
                                        //simpan ke dalam rekap presensi
                                        stat.execute("insert into rekap_presensi values "+
                                                "('"+rs2.getString(1) +"','"+rs2.getString(2) +"','"+rs2.getString(3) +
                                                "','"+rs2.getString(4) +"','"+rs2.getString(5) +"','"+rs2.getString(6) +
                                                "','"+rs2.getString(7) +"','','')");
                                        stat.execute("delete from temporary_presensi where id='"+Idpresensi.getText()+"'");
                                        //jamdatang.setText("");
                                        //jampulang.setText("");
                                    }catch (Exception er) {         
                                        System.out.println(er);  
                                        JOptionPane.showMessageDialog(null,"Maaf, Anda sudah presensi Pulang ...!");
                                    }  
                                }
                            }catch (Exception er) {                                
                                System.out.println(er);  
                            }                               
                        }
                    tampil("");
                    emptTeks();
                }else if(rs.getRow()==0){
                    JOptionPane.showMessageDialog(null,"Maaf, Gagal Presensi. ID User atau password ada yang salah ...!");
                }
            } catch (SQLException ex) {
                System.out.println(ex);   
            }  
        }

    public void tampil(String key) {
        String sql="select  pegawai.id, "+
                "pegawai.nik, "+
                "pegawai.nama, "+
                "temporary_presensi.shift,"+
                "temporary_presensi.jam_datang, "+
                "temporary_presensi.jam_pulang, "+
                "temporary_presensi.status, "+
                "temporary_presensi.keterlambatan, "+
                "temporary_presensi.durasi "+
                " from pegawai inner join temporary_presensi on pegawai.id=temporary_presensi.id order by temporary_presensi.jam_datang";
        prosesCari(sql);
    }

    private void prosesCari(String sql) {
        Valid.tabelKosong(tabMode);
        try{
            java.sql.Statement stat=connect.createStatement();
            ResultSet rs=stat.executeQuery(sql);
            while(rs.next()){
                String[] data={rs.getString(2),
                               rs.getString(3),
                               rs.getString(4),
                               rs.getString(5),
                               rs.getString(6),
                               rs.getString(7),
                               rs.getString(8),
                               rs.getString(9)};
                tabMode.addRow(data);
             }
            rs.close();
            stat.close();
        }catch(SQLException e){
            System.out.println("Error : "+e);
        }
        LCount.setText(""+tabMode.getRowCount());

    }
