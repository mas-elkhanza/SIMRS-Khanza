            Sequel.queryu("delete from temporary");

                                tabMode.getValueAt(i,2).toString()+"','"+
                                tabMode.getValueAt(i,3).toString()+"','"+
                                tabMode.getValueAt(i,4).toString()+"','"+
                                tabMode.getValueAt(i,5).toString()+"','"+
                                kdrek.getText()+", "+nmrek.getText()+"','"+Tahun.getSelectedItem()+"','','','','','','','','',''","Keuangan");

        }

    private void kdrekKeyPressed(java.awt.event.KeyEvent evt) {//GEN-FIRST:event_kdrekKeyPressed
        if(evt.getKeyCode()==KeyEvent.VK_PAGE_DOWN){
            kdrek.setText(rekening.getTextField().getText());
            Sequel.cariIsi("select nm_rek from rekening where kd_rek='"+kdrek.getText()+"'",nmrek);
        }else if(evt.getKeyCode()==KeyEvent.VK_PAGE_UP){
            kdrek.setText(rekening.getTextField().getText());
            Sequel.cariIsi("select nm_rek from rekening where kd_rek='"+kdrek.getText()+"'",nmrek);
            Tahun.requestFocus();
        }else if(evt.getKeyCode()==KeyEvent.VK_ENTER){
            kdrek.setText(rekening.getTextField().getText());
            Sequel.cariIsi("select nm_rek from rekening where kd_rek='"+kdrek.getText()+"'",nmrek);
            BtnKeluar.requestFocus();
        }
    }//GEN-LAST:event_kdrekKeyPressed

    public void prosesCari() {
       Valid.tabelKosong(tabMode);
       String thn=" rekeningtahun.thn='"+Tahun.getSelectedItem()+"' ";               
       try{
            java.sql.Statement stat=koneksi.createStatement();
            java.sql.Statement stat2=koneksi.createStatement();
            ResultSet rs=stat.executeQuery("select jurnal.tgl_jurnal,jurnal.no_jurnal,detailjurnal.debet,detailjurnal.kredit "+
                            "from jurnal inner join detailjurnal on jurnal.no_jurnal=detailjurnal.no_jurnal  where "+
                            "detailjurnal.kd_rek='"+kdrek.getText()+"' and jurnal.tgl_jurnal like '%"+Tahun.getSelectedItem()+"%' ");
            
            ResultSet rs2=stat2.executeQuery("select rekeningtahun.saldo_awal,rekening.tipe, "+
                            " rekening.balance from rekeningtahun inner join rekening on rekeningtahun.kd_rek=rekening.kd_rek "+
                            " where rekeningtahun.kd_rek='"+kdrek.getText()+"' and rekeningtahun.thn like '%"+Tahun.getSelectedItem()+"%'  ");
            double saldoakhir = 0,saldoawal=0,tampawal=0;
            String tipe="",balance="";
            while(rs2.next()){
                 saldoawal=rs2.getDouble(1);    
                 saldoakhir=rs2.getDouble(1); 
                 tipe=rs2.getString(2);
                 balance=rs2.getString(3);
            }          
                
            while(rs.next()){  
                if(tipe.equals("R")&&balance.equals("K")){
                        saldoakhir=saldoakhir+rs.getDouble(4);
                }else{
                    saldoakhir=saldoakhir+rs.getDouble(3)-rs.getDouble(4);
                }
                
                if(saldoawal<0){
                        saldoawal=saldoawal*(-1);
                }
                
                tabMode.addRow(new Object[]{rs.getString(1),
                               rs.getString(2),
                               df2.format(saldoawal),
                               df2.format(rs.getDouble(3)),
                               df2.format(rs.getDouble(4)),
                               df2.format(saldoakhir)});
                saldoawal=saldoawal+rs.getDouble(3)-rs.getDouble(4);
                //tampawal=saldoawal;
                //saldoakhir=saldoawal;
                
            }
        }catch(SQLException e){
            System.out.println("Error : "+e);
        }
    }
