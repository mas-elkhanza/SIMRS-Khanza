    private void BtnPrintActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_BtnPrintActionPerformed
        this.setCursor(Cursor.getPredefinedCursor(Cursor.WAIT_CURSOR));
        if(tabMode.getRowCount()==0){
            JOptionPane.showMessageDialog(null,"Maaf, data sudah habis. Tidak ada data yang bisa anda print...!!!!");
            //TCari.requestFocus();
        }else if(tabMode.getRowCount()!=0){
            Sequel.queryu("delete from temporary");
            int row=tabMode.getRowCount();
            for(int r=0;r<row;r++){  
                Sequel.menyimpan("temporary","'0','"+
                                tabMode.getValueAt(r,0).toString().replaceAll("'","`") +"','"+
                                tabMode.getValueAt(r,1).toString().replaceAll("'","`")+"','"+
                                tabMode.getValueAt(r,2).toString().replaceAll("'","`")+"','"+
                                tabMode.getValueAt(r,3).toString().replaceAll("'","`")+"','"+
                                tabMode.getValueAt(r,4).toString().replaceAll("'","`")+"','"+
                                tabMode.getValueAt(r,5).toString().replaceAll("'","`")+"','','','','','','','','','','',''","Rekap Harian BulananDokter"); 
            }
            Valid.MyReport("rptRBTindakanParamedis.jrxml","report","[ Rekap Bulanan Tindakan Paramedis ]",
                "select no, temp1, temp2, temp3, temp4, temp5, temp6, temp7, temp8, temp9, temp10, temp11, temp12, temp13, temp14 from temporary order by no asc");
        }
        this.setCursor(Cursor.getDefaultCursor());
    }//GEN-LAST:event_BtnPrintActionPerformed

    public void prosesCari() {
       Valid.tabelKosong(tabMode);      
       try{   
            ResultSet rs=koneksi.createStatement().executeQuery(
                    "select nip,nama from petugas where nip like '%"+kdbangsal.getText()+"%' order by nama");
            i=1;
            totaljm=0;
            while(rs.next()){
               Object[] data={""+i+".",rs.getString("nama"),"","","",""}; 
               tabMode.addRow(data);   
               jm=0;
               //rawat jalan
               ResultSet rs2=koneksi.createStatement().executeQuery(
                       "select jns_perawatan.nm_perawatan,jns_perawatan.tarif_tindakanpr,"+
                       "count(jns_perawatan.nm_perawatan) as jml,"+
                       "jns_perawatan.tarif_tindakanpr * count(jns_perawatan.nm_perawatan) as total "+
                       "from reg_periksa inner join jns_perawatan inner join rawat_jl_pr "+
                       "on rawat_jl_pr.no_rawat=reg_periksa.no_rawat "+
                       "and rawat_jl_pr.kd_jenis_prw=jns_perawatan.kd_jenis_prw "+
                       "where reg_periksa.tgl_registrasi between '"+Tgl1.getSelectedItem()+
                       "' and '"+Tgl2.getSelectedItem()+"' and rawat_jl_pr.nip='"+rs.getString("nip")+
                       "' and jns_perawatan.total_byrpr>0 group by jns_perawatan.nm_perawatan order by jns_perawatan.nm_perawatan");
               if(rs2.next()){
                    Object[] data2={"","","Rawat Jalan :","","",""};  
                    tabMode.addRow(data2);   
               }
               rs2.beforeFirst();
               a=1;
               while(rs2.next()){
                   Object[] data2={
                       "","",a+". "+rs2.getString("nm_perawatan")," "+Valid.SetAngka(rs2.getDouble("tarif_tindakanpr")),
                       rs2.getString("jml"),Valid.SetAngka(rs2.getDouble("total"))
                   }; 
                   tabMode.addRow(data2);                   
                   a++;
                   jm=jm+rs2.getDouble("total");
               }
               //rs2.close();
               
               //periksa lab
               ResultSet rs4=koneksi.createStatement().executeQuery(
                       "select jns_perawatan_lab.nm_perawatan,jns_perawatan_lab.tarif_tindakan_petugas,"+
                       "count(jns_perawatan_lab.nm_perawatan) as jml, " +
                       "jns_perawatan_lab.tarif_tindakan_petugas * count(jns_perawatan_lab.nm_perawatan) as total "+
                       " from jns_perawatan_lab inner join periksa_lab "+
                       "on periksa_lab.kd_jenis_prw=jns_perawatan_lab.kd_jenis_prw "+
                       "where periksa_lab.tgl_periksa between '"+Tgl1.getSelectedItem()+
                       "' and '"+Tgl2.getSelectedItem()+"' and periksa_lab.nip='"+rs.getString("nip")+
                       "' group by jns_perawatan_lab.nm_perawatan order by jns_perawatan_lab.nm_perawatan  ");
               if(rs4.next()){
                    Object[] data6={"","","Periksa Lab : ","","",""}; 
                    tabMode.addRow(data6); 
               }
               rs4.beforeFirst();
               a=1;
               while(rs4.next()){
                   Object[] data4={
                       "","",a+". "+rs4.getString("nm_perawatan")," "+Valid.SetAngka(rs4.getDouble("tarif_tindakan_petugas")),
                       rs4.getString("jml"),Valid.SetAngka(rs4.getDouble("total"))
                   }; 
                   tabMode.addRow(data4);                   
                   a++;
                   jm=jm+rs4.getDouble("total");
               }
               //rs4.close();  
               
               //rawat inap
               ResultSet rs3=koneksi.createStatement().executeQuery(
                       "select jns_perawatan_inap.nm_perawatan,jns_perawatan_inap.tarif_tindakanpr,"+
                       "count(jns_perawatan_inap.nm_perawatan) as jml, " +
                       "jns_perawatan_inap.tarif_tindakanpr * count(jns_perawatan_inap.nm_perawatan) as total "+
                       "from jns_perawatan_inap inner join rawat_inap_pr "+
                       "on rawat_inap_pr.kd_jenis_prw=jns_perawatan_inap.kd_jenis_prw "+
                       "where rawat_inap_pr.tgl_perawatan between '"+Tgl1.getSelectedItem()+
                       "' and '"+Tgl2.getSelectedItem()+"' and rawat_inap_pr.nip='"+rs.getString("nip")+
                       "' and jns_perawatan_inap.total_byrpr>0 group by jns_perawatan_inap.nm_perawatan order by jns_perawatan_inap.nm_perawatan  ");
               
               ResultSet rs5=koneksi.createStatement().executeQuery(
                       "select paket_operasi.nm_perawatan,operasi.biayaasisten_operator1,"+
                       "count(operasi.kode_paket) as jml, " +
                       "sum(operasi.biayaasisten_operator1) as total "+
                       "from paket_operasi inner join operasi "+
                       "on operasi.kode_paket=paket_operasi.kode_paket "+
                       "where operasi.tgl_operasi between '"+Tgl1.getSelectedItem()+
                       "' and '"+Tgl2.getSelectedItem()+"' and operasi.asisten_operator1='"+rs.getString("nip")+
                       "' and operasi.biayaasisten_operator1>0 group by operasi.kode_paket order by paket_operasi.nm_perawatan  ");  
               
               ResultSet rs6=koneksi.createStatement().executeQuery(
                       "select paket_operasi.nm_perawatan,operasi.biayaasisten_operator2,"+
                       "count(operasi.kode_paket) as jml, " +
