    private void BtnPrintActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_BtnPrintActionPerformed
        this.setCursor(Cursor.getPredefinedCursor(Cursor.WAIT_CURSOR));
        BtnCariActionPerformed(evt);
        if(tabMode.getRowCount()==0){
            JOptionPane.showMessageDialog(null,"Maaf, data sudah habis. Tidak ada data yang bisa anda print...!!!!");
            TCari.requestFocus();
        }else if(tabMode.getRowCount()!=0){
            Sequel.queryu("delete from temporary");
            int row=tabMode.getRowCount();
            for(int i=0;i<row;i++){  
                Sequel.menyimpan("temporary","'0','"+

    public void tampil() {
        String tanggal=" jurnal.tgl_jurnal between '"+TglJurnal1.getSelectedItem()+"' and '"+TglJurnal2.getSelectedItem()+"' "+
                       " and jenis='"+Jenis.getSelectedItem().toString().substring(0,1)+"' ";
        String nojur="",rek="";
        if(!NoJur.getText().equals("")){
            nojur=" and  jurnal.no_jurnal='"+NoJur.getText()+"' ";
        }        
        if(!nmrek.getText().equals("")){
            rek=" and rekening.nm_rek='"+nmrek.getText()+"' ";
        }
        
        String sql="select jurnal.no_jurnal, jurnal.no_bukti, jurnal.tgl_jurnal, jurnal.jenis, jurnal.keterangan,"+
                   "detailjurnal.kd_rek,rekening.nm_rek from jurnal inner join detailjurnal inner join rekening  on "+
                   "detailjurnal.no_jurnal=jurnal.no_jurnal and detailjurnal.kd_rek=rekening.kd_rek "+
                    " where "+tanggal+nojur+rek+" and jurnal.no_jurnal like '%"+TCari.getText()+"%' or "+
                    tanggal+nojur+rek+" and jurnal.no_bukti like '%"+TCari.getText()+"%' or "+
                    tanggal+nojur+rek+" and jurnal.keterangan like '%"+TCari.getText()+"%' or "+
                    tanggal+nojur+rek+" and detailjurnal.kd_rek like '%"+TCari.getText()+"%' or "+
                    tanggal+nojur+rek+" and rekening.nm_rek like '%"+TCari.getText()+"%' "+
                    " group by jurnal.no_jurnal order by jurnal.tgl_jurnal,jurnal.no_jurnal ";
        prosesCari(sql);
    }

    private void prosesCari(String sql) {
       Valid.tabelKosong(tabMode);
        try{
            java.sql.Statement stat=koneksiDB.condb().createStatement();
            java.sql.Statement stat2=koneksiDB.condb().createStatement();
            ResultSet rs=stat.executeQuery(sql);
            ttldebet=0;ttlkredit=0;
            while(rs.next()){
                String jns="Umum";
                switch (rs.getString(4)) {
                    case "U":
                        jns="Umum";
                        break;
                    case "P":
                        jns="Penyelesaian";
                        break;
                }
                String[] data={rs.getString(1),
                               rs.getString(2),
                               rs.getString(3),
                               jns,
                               rs.getString(5),
                               "Posting Jurnal :","",""};
                tabMode.addRow(data);
                String rek="";
                if(!nmrek.getText().equals("")){
                    rek=" and rekening.nm_rek='"+nmrek.getText()+"' ";
                }
                ResultSet rs2=stat2.executeQuery("select detailjurnal.kd_rek,rekening.nm_rek,detailjurnal.debet,detailjurnal.kredit "+
                        " from detailjurnal inner join rekening "+
                        " on detailjurnal.kd_rek=rekening.kd_rek where "+
                        " detailjurnal.no_jurnal='"+rs.getString(1)+"' "+rek+" and detailjurnal.kd_rek like '%"+TCari.getText()+"%' or "+
                        " detailjurnal.no_jurnal='"+rs.getString(1)+"' "+rek+" and rekening.nm_rek like '%"+TCari.getText()+"%'  order by detailjurnal.kd_rek  ");
                subttldebet=0;subttlkredit=0;
                int no=1;
                while(rs2.next()){
                    ttldebet=ttldebet+rs2.getDouble(3);
                    subttldebet=subttldebet+rs2.getDouble(3);
                    ttlkredit=ttlkredit+rs2.getDouble(4);
                    subttlkredit=subttlkredit+rs2.getDouble(4);
                    String[] data2={"","","","","",no+". "+rs2.getString(1)+", "+rs2.getString(2),df2.format(rs2.getDouble(3)),df2.format(rs2.getDouble(4))};
                    tabMode.addRow(data2);
                    no++;
                }
                String[] data3={"Total : ","","","","","",df2.format(subttldebet),df2.format(subttlkredit)};
                tabMode.addRow(data3);                
            }                   
            debet.setText(df2.format(ttldebet));
            kredit.setText(df2.format(ttlkredit));
        }catch(SQLException e){
            System.out.println("Error : "+e);
        }
        
    }
