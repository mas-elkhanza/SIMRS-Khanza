    private void tbKamarKeyPressed(java.awt.event.KeyEvent evt) {//GEN-FIRST:event_tbKamarKeyPressed
        if(tabMode.getRowCount()!=0){
            if(evt.getKeyCode()==KeyEvent.VK_ENTER){
                try {
                    getData();
                    if(tbKamar.getSelectedColumn()==1){
                        TCari.setText("");
                        TCari.requestFocus();
                    }
                } catch (java.lang.NullPointerException e) {
                }
            }else if((evt.getKeyCode()==KeyEvent.VK_UP)||(evt.getKeyCode()==KeyEvent.VK_DOWN)){
                try {
                    getData();
                } catch (java.lang.NullPointerException e) {
                }
            }else if(evt.getKeyCode()==KeyEvent.VK_DELETE){
                if(tbKamar.getSelectedRow()!= -1){
                    tabMode.setValueAt("", tbKamar.getSelectedRow(),0);
                }
            }           
        }
}//GEN-LAST:event_tbKamarKeyPressed

private void BtnSimpanActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_BtnSimpanActionPerformed
        if(TNoRw.getText().trim().equals("")||TKdPny.getText().trim().equals("")){
            Valid.textKosong(TCari,"Data");
        }else if(bangsal.equals("")){
            Valid.textKosong(TCari,"Lokasi");
        }else{
            try {      
                koneksi.setAutoCommit(false);
                PreparedStatement ps= koneksi.prepareStatement("insert into resep_pulang values(?,?,?,?,?,?)");
                PreparedStatement ps2= koneksi.prepareStatement("insert into gudangbarang values(?,?,?)");
                PreparedStatement ps3= koneksi.prepareStatement("update gudangbarang set stok=stok-? where kode_brng=? and kd_bangsal='"+bangsal+"'");
                for(i=0;i<tabMode.getRowCount();i++){ 
                    if(Valid.SetAngka(tabMode.getValueAt(i,0).toString())>0){
                            ps.setString(1,TNoRw.getText());
                            ps.setString(2,tabMode.getValueAt(i,1).toString());
                            ps.setString(3,tabMode.getValueAt(i,0).toString());
                            ps.setString(4,tabMode.getValueAt(i,6).toString());
                            ps.setDouble(5,Double.parseDouble(tabMode.getValueAt(i,6).toString())*Double.parseDouble(tabMode.getValueAt(i,0).toString()));
                            ps.setString(6,tabMode.getValueAt(i,4).toString());
                            ps.executeUpdate();  
                            try{
                               ps2.setString(1,tbKamar.getValueAt(i,1).toString());
                               ps2.setString(2,bangsal);

                            }

                }

                dispose();
            } catch (NumberFormatException | SQLException ex) {
                System.out.println(ex);
                JOptionPane.showMessageDialog(null,"Maaf, gagal menyimpan data. Kemungkinan ada data yang sama dimasukkan sebelumnya...!");
            }
        }

    private void prosesCari(String sql) {
        jml=0;
        for(i=0;i<tabMode.getRowCount();i++){
            if(!tabMode.getValueAt(i,0).toString().equals("")){
                jml++;
            }
        }
        
        jumlah=null;
        jumlah=new double[jml];
        stok=null;
        stok=new double[jml];
        harga=null;
        harga=new double[jml];
        kodebarang=null;
        kodebarang=new String[jml];
        namabarang=null;
        namabarang=new String[jml];
        kodesatuan=null;
        kodesatuan=new String[jml];
        letakbarang=null;
        letakbarang=new String[jml];
        namajenis=null;
        namajenis=new String[jml];
        dosis=null;
        dosis=new String[jml];
        
        index=0;        
        for(i=0;i<tabMode.getRowCount();i++){
            if(!tabMode.getValueAt(i,0).toString().equals("")){
                jumlah[index]=Double.parseDouble(tabMode.getValueAt(i,0).toString());
                kodebarang[index]=tabMode.getValueAt(i,1).toString();
                namabarang[index]=tabMode.getValueAt(i,2).toString();
                kodesatuan[index]=tabMode.getValueAt(i,3).toString();
                dosis[index]=tabMode.getValueAt(i,4).toString();
                letakbarang[index]=tabMode.getValueAt(i,5).toString();
                harga[index]=Double.parseDouble(tabMode.getValueAt(i,6).toString());
                stok[index]=Double.parseDouble(tabMode.getValueAt(i,7).toString());
                namajenis[index]=tabMode.getValueAt(i,8).toString();
                index++;
            }
        }
        
        Valid.tabelKosong(tabMode);
        
        for(i=0;i<jml;i++){
            Object[] data={jumlah[i],kodebarang[i],namabarang[i],
                           kodesatuan[i],dosis[i],letakbarang[i],harga[i],stok[i],namajenis[i]};
            tabMode.addRow(data);
        }
        
        try{
            PreparedStatement ps=koneksi.prepareStatement(sql);
            ps.setString(1,"%"+TCari.getText().trim()+"%");
            ps.setString(2,"%"+TCari.getText().trim()+"%");
            ps.setString(3,"%"+TCari.getText().trim()+"%");
            ResultSet rs=ps.executeQuery();
            while(rs.next()){
                tabMode.addRow(new Object[]{"",
                               rs.getString("kode_brng"),
                               rs.getString("nama_brng"),
                               rs.getString("kode_sat"),
                               "",
                               rs.getString("letak_barang"),
                               rs.getDouble("harga"),
                               rs.getDouble("stok"),
                               rs.getString("nama")});
            }       
            //rs.close();
            //ps.close();
        }catch(SQLException e){
            System.out.println("Error : "+e);
        }
    }

    private void setObatPasien(){
        Valid.tabelKosong(tabMode);

                " from databarang inner join jenis inner join resep_pulang on databarang.kdjns=jenis.kdjns  "+
                " and databarang.kode_brng=resep_pulang.kode_brng where resep_pulang.no_rawat=? order by databarang.nama_brng";

                               rs.getDouble("harga"),
                               rs.getDouble("stok"),
                               rs.getString("nama")});
            }     
            //rs.close();
            //ps.close();
        }catch(SQLException e){

        }
    }
