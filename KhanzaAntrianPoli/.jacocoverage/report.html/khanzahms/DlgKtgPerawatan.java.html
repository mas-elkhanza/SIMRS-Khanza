    private void tbKategoriMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_tbKategoriMouseClicked
        if(tabMode.getRowCount()!=0){
            try {
                getData();
            } catch (java.lang.NullPointerException e) {
            }
            if((var.getStatus()==true)&&(evt.getClickCount()==2)){
                dispose();
            }
        }
}//GEN-LAST:event_tbKategoriMouseClicked

    private void tbKategoriKeyPressed(java.awt.event.KeyEvent evt) {//GEN-FIRST:event_tbKategoriKeyPressed
        if(tabMode.getRowCount()!=0){
            if((evt.getKeyCode()==KeyEvent.VK_ENTER)||(evt.getKeyCode()==KeyEvent.VK_UP)||(evt.getKeyCode()==KeyEvent.VK_DOWN)){
                try {
                    getData();
                } catch (java.lang.NullPointerException e) {
                }
            }
            
            if((var.getStatus()==true)&&(evt.getKeyCode()==KeyEvent.VK_SPACE)){
                dispose();
            }
        }
}//GEN-LAST:event_tbKategoriKeyPressed

    public void tampil(){
        Valid.tabelKosong(tabMode);
        try{
            ps.setString(1,"%"+TCari.getText().trim()+"%");
            ps.setString(2,"%"+TCari.getText().trim()+"%");
            rs=ps.executeQuery();
            while(rs.next()){
                tabMode.addRow(new Object[]{rs.getString(1),rs.getString(2)});
             }
        }catch(SQLException e){
            System.out.println("Error : "+e);
        }
        LCount.setText(""+tabMode.getRowCount());
    }

    private void getData() {
        int row=tbKategori.getSelectedRow();
        if(row!= -1){
            TKd.setText(tbKategori.getValueAt(row,0).toString());
            TNm.setText(tbKategori.getValueAt(row,1).toString());
        }
    }

    public void emptTeks() {
        TKd.setText("");
        TNm.setText("");
        TCari.setText("");
        TKd.requestFocus();
        Valid.autoNomer(tabMode,"KP",3,TKd);
    }

        return TKd;
