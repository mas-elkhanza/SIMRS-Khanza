    private void BtnPrintActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_BtnPrintActionPerformed
        this.setCursor(Cursor.getPredefinedCursor(Cursor.WAIT_CURSOR));
        if(tabMode.getRowCount()==0){
            JOptionPane.showMessageDialog(null,"Maaf, data sudah habis. Tidak ada data yang bisa anda print...!!!!");
            BtnBatal.requestFocus();
        }else if(tabMode.getRowCount()!=0){            
            Valid.MyReport("rptPenyakit.jrxml","report","::[ Data Penyakit ]::","select kd_penyakit,nm_penyakit,ciri_ciri,keterangan, "+
                "nm_kategori,ciri_umum "+
                "from kategori_penyakit inner join penyakit "+
                "on penyakit.kd_ktg=kategori_penyakit.kd_ktg where  "+
                " kd_penyakit like '%"+TCari.getText().trim()+"%' or "+
                " nm_penyakit like '%"+TCari.getText().trim()+"%' or "+
                " ciri_ciri like '%"+TCari.getText().trim()+"%' or "+
                " keterangan like '%"+TCari.getText().trim()+"%' or "+
                " nm_kategori like '%"+TCari.getText().trim()+"%' or "+
                " ciri_umum like '%"+TCari.getText().trim()+"%' "+
                "order by kd_penyakit LIMIT "+awal+",500");
        }
        this.setCursor(Cursor.getDefaultCursor());
}//GEN-LAST:event_BtnPrintActionPerformed

    public void tampil() {
        Valid.tabelKosong(tabMode);
        try{
            awal="0";
            if(cmbHlm.getItemCount()>0){
                awal=hlm[Integer.parseInt(cmbHlm.getSelectedItem().toString())];
            }
            ps.setString(1,"%"+TCari.getText().trim()+"%");
            ps.setString(2,"%"+TCari.getText().trim()+"%");
            ps.setString(3,"%"+TCari.getText().trim()+"%");
            ps.setString(4,"%"+TCari.getText().trim()+"%");
            ps.setString(5,"%"+TCari.getText().trim()+"%");
            ps.setString(6,"%"+TCari.getText().trim()+"%");
            ps.setInt(7,Integer.parseInt(awal));
            rs=ps.executeQuery();
            while(rs.next()){
                tabMode.addRow(new String[] {rs.getString(1),
                               rs.getString(2),
                               rs.getString(3),
                               rs.getString(4),
                               rs.getString(5),
                               rs.getString(6)});
            }
            
            cmbHlm.removeAllItems();
            ps2.setString(1,"%"+TCari.getText().trim()+"%");
            ps2.setString(2,"%"+TCari.getText().trim()+"%");
            ps2.setString(3,"%"+TCari.getText().trim()+"%");
            ps2.setString(4,"%"+TCari.getText().trim()+"%");
            ps2.setString(5,"%"+TCari.getText().trim()+"%");
            ps2.setString(6,"%"+TCari.getText().trim()+"%");
            rs2=ps2.executeQuery();
            jumlah=0;
            if(rs2.next()){
               jumlah=rs2.getDouble("jumlah");   
            }
            x=jumlah/499;
            i=Math.ceil(x);
            z=(int) i;
            
            hlm=new String[z+1];
            for(j=1;j<=i;j++){
                 mulai=((j-1)*499+j)-1;
                 hlm[j]=Integer.toString(mulai);
                 cmbHlm.addItem(j);
            }
        }catch(SQLException e){
            System.out.println("Error : "+e);
        }
        LCount.setText(""+tabMode.getRowCount());
    }

    public void emptTeks() {
        TKd.setText("");
        TNm.setText("");
        TCiri.setText("");
        TKeterangan.setText("");
        kd_ktg.setText("");
        TKtg.setText("");
        Valid.autoNomer("penyakit","P",9,TKd);
        TKd.requestFocus();
    }

    private void getData() {
        int row=tbPenyakit.getSelectedRow();
        if(row!= -1){
            TKd.setText(tbPenyakit.getValueAt(row,0).toString());
            TNm.setText(tbPenyakit.getValueAt(row,1).toString());
            TCiri.setText(tbPenyakit.getValueAt(row,2).toString());
            TKeterangan.setText(tbPenyakit.getValueAt(row,3).toString());
            TKtg.setText(tbPenyakit.getValueAt(row,4).toString());
            Sequel.cariIsi("select kd_ktg from penyakit where kd_penyakit=?", kd_ktg,tbPenyakit.getValueAt(row,0).toString());
        }
    }

    private void isForm(){
        if(ChkInput.isSelected()==true){
            ChkInput.setVisible(false);

            FormInput.setVisible(true);      
            ChkInput.setVisible(true);
        }else if(ChkInput.isSelected()==false){           
            ChkInput.setVisible(false);            
            PanelInput.setPreferredSize(new Dimension(WIDTH,20));
            FormInput.setVisible(false);      
            ChkInput.setVisible(true);
        }
    }
