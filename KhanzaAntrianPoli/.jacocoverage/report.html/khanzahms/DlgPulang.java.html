    private void BtnCariActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_BtnCariActionPerformed
        String tgl=Sequel.cariString("select current_date()");
        String say=" rekap_presensi.jam_datang like '%"+tgl+"%' and rekap_presensi.durasi!='' ";
        prosesCari("select  pegawai.id, "+
                "pegawai.nik, "+
                "pegawai.nama, "+
                "rekap_presensi.shift, "+
                "rekap_presensi.jam_datang, "+
                "rekap_presensi.jam_pulang, "+
                "rekap_presensi.status, "+
                "rekap_presensi.keterlambatan, "+
                "rekap_presensi.durasi, "+
                "rekap_presensi.keterangan "+
                " from pegawai inner join rekap_presensi on pegawai.id=rekap_presensi.id where "+
                   " pegawai.nik like '%"+TCari.getText().trim()+"%' and "+say+
                   "or pegawai.nama like '%"+TCari.getText().trim()+"%' and "+say+
                   "or rekap_presensi.shift like '%"+TCari.getText().trim()+"%' and "+say+
                   "or rekap_presensi.status like '%"+TCari.getText().trim()+"%' and "+say+
                   "or rekap_presensi.keterlambatan like '%"+TCari.getText().trim()+"%' and "+say+
                   "or rekap_presensi.jam_datang like '%"+TCari.getText().trim()+"%' and "+say+
                   "or rekap_presensi.keterangan like '%"+TCari.getText().trim()+"%' and "+say+
                   "or rekap_presensi.jam_pulang like '%"+TCari.getText().trim()+"%' and "+say+" order by rekap_presensi.jam_datang ");
}//GEN-LAST:event_BtnCariActionPerformed

            TCari.requestFocus();
        }else if(id_pegawai.getText().trim().equals("")){

        }else if(! id_pegawai.getText().trim().equals("")){
            
            //LabelKon.setVisible(true);
            DlgInput.setLocationRelativeTo(null);
            DlgInput.setVisible(true);
        }

            Valid.textKosong(catatan,"Pegawai");

            Valid.textKosong(id_pegawai,"Jam Datang");

            tampil("");

        }

    public void tampil(String key) {
        String tgl=Sequel.cariString("select current_date()");
        String sql="select  pegawai.id, "+
                "pegawai.nik, "+
                "pegawai.nama, "+
                "rekap_presensi.shift, "+
                "rekap_presensi.jam_datang, "+
                "rekap_presensi.jam_pulang, "+
                "rekap_presensi.status, "+
                "rekap_presensi.keterlambatan, "+
                "rekap_presensi.durasi, "+
                "rekap_presensi.keterangan "+
                " from pegawai inner join rekap_presensi on pegawai.id=rekap_presensi.id where rekap_presensi.jam_datang like '%"+tgl+"%' order by rekap_presensi.jam_datang";
        prosesCari(sql);
    }

    private void prosesCari(String sql) {
        Valid.tabelKosong(tabMode);
        try{
            java.sql.Statement stat=connect.createStatement();
            ResultSet rs=stat.executeQuery(sql);
            
            rs.last();
            primary=new String[rs.getRow()];
            rs.beforeFirst();
            int i=0;
            
            while(rs.next()){
                String[] data={rs.getString(2),
                               rs.getString(3),
                               rs.getString(4),
                               rs.getString(5),
                               rs.getString(6),
                               rs.getString(7),
                               rs.getString(8),
                               rs.getString(9),
                               rs.getString(10)};
                tabMode.addRow(data);
                primary[i]=rs.getString(1);
                i++;
             }
            rs.close();
            stat.close();
        }catch(SQLException e){
            System.out.println("Error : "+e);
        }
        LCount.setText(""+tabMode.getRowCount());

    }


    private void getData() {
        int row=tbPegawai.getSelectedRow();
        if(row!= -1){
            id_pegawai.setText(primary[row]);
            jam_datang.setText(tabMode.getValueAt(row,3).toString());
            catatan.setText(tabMode.getValueAt(row,8).toString());
            
        }
    }
    
    public void emptTeks() {
        catatan.setText("");
        id_pegawai.setText("");
        jam_datang.setText("");
    }
