    private void BtnSimpanActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_BtnSimpanActionPerformed
        if(distrib.getText().trim().equals("")){
            Valid.textKosong(distrib,"Harga Distributor");
        }else if(grosir.getText().trim().equals("")){
            Valid.textKosong(grosir,"Harga Grosir");
        }else if(retail.getText().trim().equals("")){
            Valid.textKosong(retail,"Harga Retail");
        }else if(tabMode.getRowCount()==0){
            Sequel.menyimpan("setpenjualan","'"+distrib.getText()+"','"+grosir.getText()+"','"+retail.getText()+"'","Set Harga");
            tampil();
            emptTeks();
        }else if(tabMode.getRowCount()>0){
            JOptionPane.showMessageDialog(null,"Maaf, Hanya diijinkan satu record ...!!!!");
            distrib.requestFocus();
        }
}//GEN-LAST:event_BtnSimpanActionPerformed

    private void BtnHapusActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_BtnHapusActionPerformed

            JOptionPane.showMessageDialog(null,"Maaf, data sudah habis...!!!!");
            distrib.requestFocus();
        }else if(grosir.getText().trim().equals("")){
            JOptionPane.showMessageDialog(null,"Maaf, Gagal menghapus. Pilih dulu data yang mau dihapus.\nKlik data pada table untuk memilih...!!!!");
        }else if(! grosir.getText().trim().equals("")){
            Sequel.queryu("delete from setpenjualan");
            tampil();
            emptTeks();
        }
}//GEN-LAST:event_BtnHapusActionPerformed

    public void tampil() {
        String sql="select distributor,grosir,retail from setpenjualan";
        prosesCari(sql);
    }

    private void prosesCari(String sql) {
        Valid.tabelKosong(tabMode);
        try{
            java.sql.Statement stat=koneksi.createStatement();
            ResultSet rs=stat.executeQuery(sql);
            while(rs.next()){
                tabMode.addRow(new Object[]{rs.getString(1),rs.getString(2),rs.getString(3)});
             }
        }catch(SQLException e){
            System.out.println("Error : "+e);
        }
    }

    private void getData() {
        int row=tbAdmin.getSelectedRow();
        if(row!= -1){
            distrib.setText(tabMode.getValueAt(row,0).toString());
            grosir.setText(tabMode.getValueAt(row,1).toString());
            retail.setText(tabMode.getValueAt(row,2).toString());
        }
    }

    public void emptTeks() {
        distrib.setText("0");
        grosir.setText("0");
        retail.setText("0");
        distrib.requestFocus();
    }
